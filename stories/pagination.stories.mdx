# Pagination 分页

### 引入组件

```js
import { Pagination } from 'graceful-ui';
```

### 基本分页
基本的分页，可以使用`current`指定当前页数,支持`.sync`修饰符, 也可以监听`on-change`事件执行其他操作
<Preview>
  <Story name="基本分页">
    {{
      components: { Pagination },
      template: `
        <div>
            <Pagination :current.sync="current" :total="total"></Pagination>
        </div>
      `,
      data(){
          return { 
              current: 1,
              total: 100
          }
      }
    }}
  </Story>
</Preview>

### 快速跳转
使用`show-elevator`参数,实现快速跳转
<Preview>
  <Story name="快速跳转">
    {{
      components: { Pagination },
      template: `
        <div>
            <Pagination :current.sync="current" :total="total" show-elevator></Pagination>
        </div>
      `,
      data(){
          return { 
              current: 1,
              total: 100
          }
      }
    }}
  </Story>
</Preview>

### 每页数量
使用`show-sizer`显示页数切换选项
<Preview>
  <Story name="每页数量">
    {{
      components: { Pagination },
      template: `
        <div style="height: 200px;">
            <Pagination :current.sync="current" :total="total" show-sizer @on-page-size-change="handlePageSizeChange"></Pagination>
        </div>
      `,
      data(){
          return { 
              current: 1,
              total: 100
          }
      },
      methods: {
        handlePageSizeChange(pageSize){
          this.$Alert({
            content: `pageSize变为: ${pageSize}`
          })
        }
      }
    }}
  </Story>
</Preview>
