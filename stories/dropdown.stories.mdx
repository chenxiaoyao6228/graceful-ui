# Dropdown 下拉组件

### 引入组件

```js
import { Dropdown, DropdownMenu, DropdownItem } from 'graceful-ui';
```

### 基本使用
配合`DropdownMenu`和`DropdownItem`两个组件来使用，并且给列表设置具名`slot`为list。

<Preview>
  <Story name="基本使用">
    {{
      template: `
        <div style="height: 200px;">
          <Dropdown
            @on-click="handleClick"
          >
            <span>
              下拉菜单
              <Icon type="arrow-down"></Icon>
            </span>
            <DropdownMenu slot="list">
              <DropdownItem>洛杉矶湖人</DropdownItem>
              <DropdownItem>纽约尼克斯</DropdownItem>
              <DropdownItem>费城76人</DropdownItem>
              <DropdownItem>芝加哥公牛</DropdownItem>
            </DropdownMenu>
          </Dropdown>
        </div>
      `,
      components: {
        Dropdown,DropdownMenu, DropdownItem, Icon
      },
      methods: {
        handleClick(options){
          this.$Alert({
            content: `被点击的item名称为: ${JSON.stringify(options)}`
          })
        }
      },
    }}
  </Story>
</Preview>
